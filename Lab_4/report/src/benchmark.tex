\subsection{Сравнение алгоритмов}

\textbf{Сортировка пузырьком vs Быстрая сортировка:}
\begin{itemize}
    \item \textbf{BubbleSort}: Время O(n²)
    \item \textbf{QuickSort}: Время O(n log n) (В СРЕДНЕМ!)
\end{itemize}

\textbf{Производная методом 1 vs методом 2:}
\begin{itemize}
    \item \textbf{Derivative1}: (f(x+Δx) - f(x)) / Δx - точность O(Δx)
    \item \textbf{Derivative2}: (f(x+Δx) - f(x-Δx)) / (2Δx) - точность O(Δx²)
\end{itemize}
